\subsection{Limitations}

\begin{frame}
\frametitle{Power supply}

  \begin{itemize}
    \item Li-Po Battery: ~14V, 5Ah $\rightarrow$ 0.055 kWh
    \item Flight time: 5min - 20min
    \item Charge time: about 2 hours
%	\item Gasoline
  \end{itemize}
  
\end{frame}
